// This object has tag "GameManager"

DEFINE CLASS GameManager EXTENDS MonoBehaviour
    DECLARE PUBLIC STRING CurrentScene
    DECLARE PRIVATE STRING CurrentLocation

    DECLARE PUBLIC INT Currency
    DECLARE PUBLIC INT NumberOfSeatsInRestaurant
    DECLARE PRIVATE INT CustomerNextSpawnTime

    DECLARE PUBLIC LIST OF GAMEOBJECTS Customers

    DECLARE PRIVATE FLOAT CustomerSpawnTimer

    DECLARE PUBLIC BOOLEAN PlayerShouldExist
    DECLARE PRIVATE BOOLEAN CustomerServedInKitchen

    DECLARE PUBLIC GAMEOBJECT MainMenuObject
    DECLARE PUBLIC GAMEOBJECT Player
    DECLARE PUBLIC GAMEOBJECT Customer

    DECLARE PUBLIC VECTOR3 PlayerJoystickPosition

    DECLARE PRIVATE VOID Start() DO
        Customers <- {}
        CustomerSpawnTimer <- 0
        NumberOfSeatsInRestaurant <- 4 // flexible
        PlayerShouldExist <- FALSE
        Currency <- 0
        CurrentScene <- MainMenu
        INSTANTIATE(MainMenuObject, CURRENT_POSITION, CURRENT_ROTATION)
        CurrentLocation <- GameConstants.KITCHEN
        CustomerNextSpawnTime <- 0
    END FUNCTION

    DECLARE PRIVATE VOID Update() DO
        IF (CurrentScene == GameConstants.LEVEL_ONE_PART_ONE OR CurrentScene == GameConstants.LEVEL_ONE_PART_TWO OR CurrentScene == GameConstants.LEVEL_TWO_PART_ONE OR CurrentScene == GameConstants.LEVEL_TWO_PART_TWO) THEN
            PlayerShouldExist <- TRUE
            IF (PlayerShouldExist AND Player.loaded == FALSE) THEN
                Player.OnLoad()
                SET PLAYER CURRENT_POSITION TO CENTRE OF KITCHEN BUT MAINTAIN CURRENT DEPTH
            END IF

            IF (CurrentLocation == GameConstants.KITCHEN) THEN
                CustomerSpawnTimer <- CustomerSpawnTimer + DELTATIME

                IF (CustomerSpawnTimer >= CustomerNextSpawnTime) THEN
                    CustomerNextSpawnTime <- RANDOM_RANGE(GameConstants.CUSTOMER_SPAWN_RATE_MIN, GameConstants.CUSTOMER_SPAWN_RATE_MAX)

                    IF (Customers.LENGTH < NumberOfSeatsInRestaurant) THEN
                        GAMEOBJECT New_Customer <- INSTANTIATE(Customer, GameConstants.CUSTOMER_SPAWNING_LOCATION, CURRENT_ROTATION)
                        Customers.ADD(New_Customer)
                    END IF
                END IF
            END IF
        ELSE
            PlayerShouldExist <- FALSE
            IF (PlayerShouldExist == FALSE AND Player.loaded) THEN
                Player.OnGarbageCollection() // Note, this should not delete the player, simply gc textures.
            END IF
        END IF
    END FUNCTION
END CLASS
