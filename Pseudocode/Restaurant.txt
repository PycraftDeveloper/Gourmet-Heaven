DEFINE CLASS Restaurant EXTENDS MonoBehaviour
    DECLARE PRIVATE BOOLEAN Loaded
    DECLARE PUBLIC GAMEOBJECT RestaurantBackground
    DECLARE PUBLIC GAMEOBJECT Table

    DECLARE PRIVATE ARRAY OF GAMEOBJECTS Tables[4]

    DECLARE PUBLIC VOID OnLoad() DO
        INSTANTIATE(RestaurantBackground, transform.position, transform.rotation)
        Loaded <- TRUE
    END FUNCTION

    DECLARE PUBLIC VOID OnGarbageCollection() DO
        Loaded <- FALSE
        DESTROY(RestaurantBackground)
    END FUNCTION

    DECLARE PRIVATE VOID AddCustomerToRestaurant(GAMEOBJECT Customer) DO
        BOOLEAN Space_Found <- FALSE
        INT Table_Index
        WHILE Space_Found IS FALSE DO
            Table_Index <- RANDOM_RANGE(0, GameConstants.NUMBER_OF_TABLES) // customers pick tables randomly
            FOR i <- 0 TO GameConstants.NUMBER_OF_SEATS_PER_TABLE DO
                IF Tables[Table_Index][i] IS NULL THEN
                    Space_Found <- TRUE
                    IF i == 0 THEN
                        Customer.Depth <- -1
                        Customer.tag <- "sitting_customer_TOP"
                        Customer.transform.position <- VECTOR3(Tables[Table_Index].transform.position.x, Tables[Table_Index].transform.position.y - Customer.SIZE.y, 0) // position relative to the table
                    ELSE IF i == 1 OR i == 2 THEN
                        Customer.Depth <- 0
                        Customer.tag <- "sitting_customer_MIDDLE"
                        IF i == 1 THEN
                            Customer.transform.position <- VECTOR3(Tables[Table_Index].transform.position.x - Customer.SIZE.x, Tables[Table_Index].transform.position.y, 0) // position relative to the table
                        ELSE
                            Customer.transform.position <- VECTOR3(Tables[Table_Index].transform.position.x + Customer.SIZE.x, Tables[Table_Index].transform.position.y, 0) // position relative to the table
                        END IF
                    ELSE
                        Customer.Depth <- 1
                        Customer.tag <- "sitting_customer_BOTTOM"
                        Customer.transform.position <- VECTOR3(Tables[Table_Index].transform.position.x, Tables[Table_Index].transform.position.y + Customer.SIZE.y, 0) // position relative to the table
                    END IF
                    Tables[Table_Index][i] <- Customer
                    BREAK
                END IF
            END FOR
        END WHILE
    END FUNCTION

    DECLARE PRIVATE VOID Start() DO
        GameManager <- GET_GAME_OBJECT_WITH_TAG("GameManager")
        Loaded <- FALSE
        GAMEOBJECT NewTable
        INT Table_X_Position <- 0
        INT Table_Y_Position <- 0
        FOR i <- 0 TO GameConstants.NUMBER_OF_TABLES DO
            NewTable <- INSTANTIATE(TABLE, VECTOR3(Table_X_Position, Table_Y_Position, 1), transform.rotation)
            Tables[i].ADD(NewTable)
        END FOR
    END FUNCTION

    DECLARE PRIVATE VOID Update() DO
    END FUNCTION
END CLASS
