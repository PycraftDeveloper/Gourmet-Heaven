// MG is a contraction of Mini-Game

DEFINE CLASS RamenMG EXTENDS MonoBehaviour
    DECLARE PRIVATE BOOLEAN Loaded
    DECLARE PUBLIC GAMEOBJECT GameManager

    DECLARE PUBLIC GAMEOBJECT MGbackgroundTexture
    DECLARE PUBLIC GAMEOBJECT RawRamenTexture
    DECLARE PRIVATE FLOAT Timer
    DECLARE PUBLIC BOOLEAN MGcomplete
    DECLARE PUBLIC STRING MGstate
    DECLARE PUBLIC GAMEOBJECT Player

    DECLARE PRIVATE VECTOR3 Bowl_Target
    DECLARE PRIVATE FLOAT Bowl_Radius

    DECLARE PUBLIC VOID OnLoad() DO
        INSTANTIATE(MGbackgroundTexture, transform.position, transform.rotation)
        INSTANTIATE(RawRamenTexture, transform.position, transform.rotation)
        Loaded <- TRUE
    END FUNCTION

    DECLARE PUBLIC VOID OnGarbageCollection() DO
        Loaded <- FALSE
        DESTROY(MGbackgroundTexture)
        DESTROY(RawRamenTexture)
    END FUNCTION

    DECLARE PRIVATE VOID Start() DO
        GameManager <- GET_GAME_OBJECT_WITH_TAG("GameManager")
        Player <- GET_GAME_OBJECT_WITH_TAG("Player")
        Loaded <- FALSE
        Timer <- 0
        MGComplete <- FALSE
        Bowl_Target <- NEW VECTOR3(0.0f, 0.0f, 0.0f) // centre of background
        Bowl_Radius <- 3 // set to radius of bowl from background texture
        MGstate <- NULL
    END FUNCTION

    DECLARE PRIVATE VOID Update() DO
        Timer <- timer + DELTATIME
        RawRamenTexture.transform.position <- NEW VECTOR3(SIN(timer) * VIEWPORT.x, CENTERED IN Y, 1) // moves in a sine motion across screen.

        IF USER_INPUT THEN // tap screen or clicked mouse
            IF RawRamenTexture.transform.position - Bowl_Target < Bowl_Radius THEN
                MGstate <- GameConstants.SUCCESS
            ELSE
                MGstate <- GameConstants.FAIL
            END IF
            Player.MadeMeal <- GameConstants.RAMEN
        END IF
    END FUNCTION
END CLASS
