// This object has tag "Player"

DEFINE CLASS Player EXTENDS MonoBehaviour
    DECLARE PRIVATE BOOLEAN Loaded

    DECLARE PUBLIC GAMEOBJECT PlayerTexture
    DECLARE PUBLIC GAMEOBJECT PlayerJoystickInput // exists when player exists
    DECLARE PUBLIC GAMEOBJECT GameManager
    DECLARE PUBLIC GAMEOBJECT Kitchen
    DECLARE PUBLIC GAMEOBJECT Restaurant

    DECLARE PUBLIC FLOAT Depth

    DECLARE PUBLIC STRING MadeMeal

    DECLARE PRIVATE VOID OnCollisionEnter(Collision collision) DO
        IF (collision.GameObject.tag == "kitchen_scene") THEN
            MOVE BACK INTO SCENE
        ELSE IF (collision.GameObject.tag == "kitchen_appliance_sushi_station" AND MadeMeal IS NULL) THEN
            FOR Customer IN GameManager.Customers DO
                IF Customer.Meal IS SUSHI THEN
                    DISPLAY UI PROMPT "play sushi minigame"
                    BREAK
                ENDIF
            END FOR
        ELSE IF (collision.GameObject.tag == "kitchen_appliance_ramen_station" AND MadeMeal IS NULL) THEN
            FOR Customer IN GameManager.Customers DO
                IF Customer.Meal IS RAMEN THEN
                    DISPLAY UI PROMPT "play ramen minigame"
                    BREAK
                ENDIF
            END FOR
        ... // extend for other mini-games as needed
        ELSE IF (collision.GameObject.tag == 'kitchen_appliance_trash_station' AND MadeMeal IS NOT NULL) THEN
            DISPLAY UI PROMPT "trash meal"
        ELSE IF (collision.GameObject.tag == "restaurant_scene") THEN
            MOVE BACK INTO SCENE
        ELSE IF (collision.GameObject.tag == "restaurant_table") THEN
            DISPLAY UI PROMPT 'serve customer'
        ELSE IF (collision.GameObject.tag == "kitchen_portal") THEN // TO restaurant FROM kitchen
            GameManager.CurrentLocation <- GameConstants.RESTAURANT

            FOR Customer IN GameManager.Customers DO // free up customer textures for those that 'persist'
                IF Customer.Location == GameConstants.KITCHEN THEN
                    Customer.OnGarbageCollection()
                END IF
            END FOR

            Kitchen.OnGarbageCollection()

        ELSE IF (collision.GameObject.tag == "restaurant_portal") THEN // TO kitchen FROM restaurant
            GameManager.CurrentLocation <- GameConstants.KITCHEN

            customers_to_destroy <- TRUE // remove customers that have been served
            WHILE (customers_to_destroy) DO
                FOR Customer IN GameManager.Customers DO
                    IF Customer.Destroy_On_Player_Exit THEN
                        DESTROY(Customer)
                        GameManager.Customers.REMOVE(Customer)
                        customers_to_destroy <- TRUE
                        BREAK
                    ELSE
                        customers_to_destroy <- FALSE
                    END IF
                END FOR
            END WHILE

            FOR Customer IN GameManager.Customers DO // free up customer textures for those that 'persist'
                IF Customer.Location == GameConstants.RESTAURANT THEN
                    Customer.OnGarbageCollection()
                END IF
            END FOR

            Restaurant.OnGarbageCollection()
        ELSE IF (collision.GameObject.tag == "sitting_customer_TOP") THEN
            Depth <- collision.GameObject.CURRENT_DEPTH_POSITION - 1 // player appears behind
        ELSE IF (collision.GameObject.tag == "sitting_customer_MIDDLE") THEN
            Depth <- 0 // player on same level as these customers
        ELSE IF (collision.GameObject.tag == "sitting_customer_BOTTOM") THEN
            Depth <- collision.GameObject.CURRENT_DEPTH_POSITION + 1 // player appears on-top
        END IF
    END FUNCTION

    DECLARE PUBLIC VOID OnLoad() DO
        INSTANTIATE(PlayerTexture, CURRENT_POSITION, CURRENT_ROTATION)
        INSTANTIATE(PlayerJoystickInput, GameManager.PlayerJoystickPosition, CURRENT_ROTATION)
        Loaded <- TRUE
    END FUNCTION

    DECLARE PUBLIC VOID OnGarbageCollection() DO
        Loaded <- FALSE
        DESTROY(PlayerTexture)
        DESTROY(PlayerJoystickInput)
    END FUNCTION

    DECLARE PRIVATE VOID Start() DO
        GameManager <- GET_GAME_OBJECT_WITH_TAG("GameManager")
        Loaded <- FALSE
        CURRENT_POSITION <- NEW VECTOR3(0.0f, 0.0f, 0.0f) // player is on layer 0
        Depth <- 0
        MadeMeal <- NULL
    END FUNCTION

    DECLARE PRIVATE VOID Update() DO
        VECTOR2 PlayerJoystickInputPosition <- PlayerJoystickInput.GetMagnitude()
        FLOAT x_position <- CURRENT_X_POSITION + PlayerJoystickInputPosition.x * GameConstants.PLAYER_MOVEMENT_SPEED * DELTATIME
        FLOAT y_position <- CURRENT_Y_POSITION + PlayerJoystickInputPosition.y * GameConstants.PLAYER_MOVEMENT_SPEED * DELTATIME
        CURRENT_POSITION <- NEW VECTOR3(x_position, y_position, Depth)

        PlayerTexture.CURRENT_POSITION <- CURRENT_POSITION;
        PlayerTexture.CURRENT_ROTATION <- CURRENT_ROTATION; // likely shouldn't be necessary in this game, but could be useful later

        IF CustomerServedInKitchen THEN
            FOR Customer IN Customers DO
                IF Customer.Location == GameConstants.KITCHEN THEN
                    Customer.CURRENT_X_POSITION <- Customer.CURRENT_X_POSITION + Customer.MODEL.GETSIZE() // shuffle down in queue
                END IF
            END FOR
        END IF
    END FUNCTION
END CLASS
