// MG is a contraction of Mini-Game

DECLARE PRIVATE CONSTANT FLOAT MIN_VELOCITY = 10.0f
DECLARE PRIVATE CONSTANT FLOAT MAX_VELOCITY = 20.0f // should be a large range

DEFINE CLASS SushiMG EXTENDS MonoBehaviour
    DECLARE PRIVATE BOOLEAN Loaded
    DECLARE PUBLIC GAMEOBJECT GameManager

    DECLARE PUBLIC GAMEOBJECT MGbackgroundTexture
    DECLARE PUBLIC GAMEOBJECT RawSushiTexture
    DECLARE PUBLIC GAMEOBJECT Player

    DECLARE PUBLIC FLOAT InputStartPosition
    DECLARE PUBLIC FLOAT InputEndPosition
    DECLARE PUBLIC FLOAT InputDuration

    DECLARE PUBLIC BOOLEAN Tapped

    DECLARE PUBLIC BOOLEAN MGcomplete
    DECLARE PUBLIC STRING MGstate

    DECLARE PUBLIC VOID OnLoad() DO
        INSTANTIATE(MGbackgroundTexture, transform.position, transform.rotation)
        INSTANTIATE(RawSushiTexture, transform.position, transform.rotation)
        Loaded <- TRUE
    END FUNCTION

    DECLARE PUBLIC VOID OnGarbageCollection() DO
        Loaded <- FALSE
        DESTROY(MGbackgroundTexture)
        DESTROY(RawSushiTexture)
    END FUNCTION

    DECLARE PRIVATE VOID Start() DO
        GameManager <- GET_GAME_OBJECT_WITH_TAG("GameManager")
        Player <- GET_GAME_OBJECT_WITH_TAG("Player")
        Loaded <- FALSE
        Tapped <- FALSE
        InputStartPosition <- 0.0f // initial default
        InputEndPosition <- 0.0f // initial default
        InputDuration <- 0.0f
        MGstate <- NULL
        MGComplete <- FALSE
    END FUNCTION

    DECLARE PRIVATE VOID Update() DO
        IF USER_INPUT THEN // tap screen or clicked mouse
            Tapped <- TRUE
            InputDuration <- InputDuration + DELTATIME
            IF InputStartPosition == 0.0f THEN
                InputStartPosition <- USER_INPUT.POSITION.x
            ELSE
                IF USER_INPUT.POSITION.x > InputEndPosition THEN
                    InputEndPosition <- USER_INPUT.position.x
                END IF
            END IF
        END IF

        IF Tapped THEN
            MGcomplete <- TRUE
            FLOAT Velocity = (InputEndPosition - InputStartPosition) / InputDuration
            IF Velocity IN RANGE(MIN_VELOCITY, MAX_VELOCITY) THEN
                MGstate <- GameConstants.SUCCESS
            ELSE
                MGstate <- GameConstants.FAIL
            END IF
            Player.MadeMeal <- GameConstants.SUSHI
        END IF
    END FUNCTION
END CLASS
