// This object has tag "kitchen_appliance_trash_station"

DEFINE CLASS TrashAppliance EXTENDS MonoBehaviour
    DECLARE PRIVATE BOOLEAN Loaded
    DECLARE PRIVATE BOOLEAN IsGlowing

    DECLARE PRIVATE GAMEOBJECT NormalApplianceTexture
    DECLARE PRIVATE GAMEOBJECT GlowingApplianceTexture

    DECLARE PUBLIC VOID OnLoad() DO
        IF IsGlowing THEN
            INSTANTIATE(GlowingApplianceTexture, CURRENT_POSITION, CURRENT_ROTATION)
        ELSE
            INSTANTIATE(NormalApplianceTexture, CURRENT_POSITION, CURRENT_ROTATION)
        END IF
        Loaded <- TRUE
    END FUNCTION

    DECLARE PUBLIC VOID OnGarbageCollection() DO
        Loaded <- FALSE
        IF IsGlowing THEN
            DESTROY(GlowingApplianceTexture)
        ELSE
            DESTROY(NormalApplianceTexture)
        END IF
    END FUNCTION

    DECLARE PUBLIC VOID SetGlowing(BOOLEAN glow) DO
        IF glow != IsGlowing THEN // switch textures
            OnGarbageCollection()
            Load()
        END IF
    END FUNCTION

    DECLARE PRIVATE VOID Start() DO
        Loaded <- FALSE
    END FUNCTION

    DECLARE PRIVATE VOID Update() DO
    END FUNCTION
END CLASS
