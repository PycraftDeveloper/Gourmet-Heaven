DECLARE CLASS PhoAppliance EXTENDS MonoBehaviour
    DECLARE PUBLIC GAMEOBJECT NormalPhoApplianceTexture
    DECLARE PUBLIC GAMEOBJECT GlowPhoApplianceTexture

    DECLARE PRIVATE BOOLEAN Loaded
    DECLARE PRIVATE BOOLEAN Glowing

    DECLARE PUBLIC VOID SetGlowing(BOOLEAN Value) DO
        IF Value EQUALS Glowing THEN
            RETURN
        END IF

        IF Loaded THEN
            GarbageCollect()
            Glowing <- Value
            Load()
        END IF
    END METHOD

    DECLARE PUBLIC VOID Load() DO
        IF Loaded THEN
            RETURN
        END IF

        IF Glowing THEN
            INSTANTIATE(GlowPhoApplianceTexture, PHO_APPLIANCE_POSITION, CURRENT_ROTATION)
        ELSE
            INSTANTIATE(NormalPhoApplianceTexture, PHO_APPLIANCE_POSITION, CURRENT_ROTATION)
        END IF
        Loaded <- TRUE
    END METHOD

    DECLARE PUBLIC VOID GarbageCollect() DO
        IF Loaded IS FALSE THEN
            RETURN
        END IF

        Loaded <- FALSE

        IF Glowing THEN
            DESTROY(GlowPhoApplianceTexture)
        ELSE
            DESTROY(NormalPhoApplianceTexture)
        END IF
    END METHOD

    DECLARE PRIVATE VOID OnStart() DO
        Loaded <- FALSE
    END METHOD

    DECLARE PRIVATE VOID Update() DO
    END METHOD
END CLASS
