// it is assumed that when customers are destroyed, they are automatically removed from any associated data structures

DECLARE CLASS Customer EXTENDS MonoBehaviour
    DECLARE PUBLIC GAMEOBJECT CustomerTexture
    DECLARE PUBLIC GAMEOBJECT CurrentLevel

    DECLARE PUBLIC STRING Meal
    DECLARE PUBLIC INT Number_Of_Meals_To_Order

    DECLARE PRIVATE BOOLEAN Loaded

    DECLARE PUBLIC STRING Location

    DECLARE PUBLIC FLOAT Depth

    DECLARE PRIVATE FLOAT Patience

    DECLARE PRIVATE INT Customer_Satisfaction

    DECLARE PUBLIC VOID Served() DO
        Patience <- RANDOM_RANGE(CUSTOMER_PATIENCE_MIN, CUSTOMER_PATIENCE_MAX)
        IF Number_Of_Meals_To_Order IS 0 THEN
            Meal <- ""
            RETURN
        ELSE
            Meal <- RANDOM_CHOICE_FROM(MEALS)
        Number_Of_Meals_To_Order <- Number_Of_Meals_To_Order - 1
        Customer_Satisfaction <- Customer_Satisfaction + 1
    END METHOD

    DECLARE PUBLIC VOID ServedWrong() DO
        Customer_Satisfaction <- Customer_Satisfaction - 2
        Served()
    END METHOD

    DECLARE PUBLIC VOID Load() DO
        IF Loaded THEN
            RETURN
        END IF

        INSTANTIATE(CustomerTexture, CURRENT_POSITION, CURRENT_ROTATION)
        Loaded <- TRUE
    END METHOD

    DECLARE PUBLIC VOID GarbageCollect() DO
        IF Loaded IS FALSE THEN
            RETURN
        END IF

        Loaded <- FALSE
        DESTROY(CustomerTexture)
    END METHOD

    DECLARE PRIVATE VOID OnStart() DO
        Loaded <- FALSE
        Customer_Satisfaction <- 0
        CurrentLevel <- GET_GAME_OBJECT_WITH_TAG("CurrentLevel)
        Meal <- RANDOM_CHOICE_FROM(MEALS)
        CurrentLevel.Customers_In_Scene <- CurrentLevel.Customers_In_Scene + 1
        Location <- KITCHEN

        IF CurrentLevel.Level IS LEVEL_ONE THEN
            IF CurrentLevel.Sub_Level IS PART_ONE THEN
                Number_Of_Meals_To_Order <- 0
            ELSE
                Number_Of_Meals_To_Order <- RANDOM_RANGE(0, 1)
            END IF
        ELSE
            IF CurrentLevel.Sub_Level IS PART_ONE THEN
                Number_Of_Meals_To_Order <- RANDOM_RANGE(0, 2)
            ELSE
                Number_Of_Meals_To_Order <- RANDOM_RANGE(0, 3)
            END IF
        END IF
    END METHOD

    DECLARE PRIVATE VOID Update() DO
        CURRENT_POSITION.z <- Depth

        IF Location IS RESTAURANT THEN
            Patience <- Patience - DELTATIME
            IF Patience <= 0 THEN
                DESTROY(THIS GAMEOBJECT) // Customer leaves if served too slowly
            END IF
        END IF
    END METHOD
END CLASS
