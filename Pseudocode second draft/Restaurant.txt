DECLARE CLASS Restaurant EXTENDS MonoBehaviour
    DECLARE PUBLIC GAMEOBJECT RestaurantBackgroundTexture
    DECLARE PUBLIC GAMEOBJECT Table
    DECLARE PUBLIC GAMEOBJECT Player

    DECLARE PUBLIC LIST OF GAMEOBJECTs Tables

    DECLARE PRIVATE BOOLEAN Loaded

    DECLARE PUBLIC VOID AddCustomerToRestaurant(GAMEOBJECT Customer) DO
        BOOLEAN Seated <- FALSE
        WHILE Seated IS FALSE DO
            GAMEOBJECT Target_Table <- RANDOM_CHOICE_FROM(Tables)
            FOR i <- 0 TO Target_Table.Seats.LENGTH DO
                IF Target_Table.Seats[i] IS NULL THEN
                    IF i IS 0 OR i IS 3 THEN
                        Customer.Depth <- Customer.Depth - 1
                    END IF
                    Target_Table.Seats[i] <- Customer
                    Seated <- TRUE
                    BREAK
                END IF
            END FOR
        END WHILE
    END METHOD

    DECLARE PUBLIC VOID Load() DO
        IF Loaded THEN
            RETURN
        END IF

        INSTANTIATE(RestaurantBackgroundTexture, CURRENT_POSITION, CURRENT_ROTATION)
        FOR i <- 0 TO 4 DO
            Tables[i].Load()
        END FOR
        Loaded <- TRUE
    END METHOD

    DECLARE PUBLIC VOID GarbageCollect() DO
        IF Loaded IS FALSE THEN
            RETURN
        END IF

        Loaded <- FALSE
        DESTROY(RestaurantBackgroundTexture)
        FOR i <- 0 TO 4 DO
            Tables[i].GarbageCollect()
        END FOR
    END METHOD

    DECLARE PRIVATE VOID OnStart() DO
        Player <- GET_GAME_OBJECT_WITH_TAG("Player")
        Loaded <- FALSE
        FOR i <- 0 TO 4 DO // load 4 tables into the Tables array
            NEW GAMEOBJECT New_Table <- INSTANTIATE(Table, CURRENT_POSITION, CURRENT_ROTATION)
            Tables.ADD(New_Table)
        END FOR
    END METHOD

    DECLARE PRIVATE VOID Update() DO
        IF Player IN COLLISION WITH RESTAURANT_SCENE THEN
            MOVE PLAYER BACK INTO SCENE
        END IF

        FOR Scene_Table IN Tables DO
            IF Player IN COLLISION WITH Scene_Table THEN
                INT index <- 0
                FOR Seated_Customer IN Scene_Table.Seats DO
                    IF Player IN COLLISION WITH Seated_Customer THEN
                        IF index IS 0 OR index IS 1 THEN
                            Player.POSITION.z <- Customer.POSITION.z - 1
                        ELSE
                            Player.POSITION.z <- Customer.POSITION.z + 1
                        END IF
                    END IF

                    IF USER INTERACTS WITH TABLE THEN
                        IF Seated_Customer.Meal IN Player.Meals THEN
                            Player.Meals.REMOVE(Served_Customer.Meal)
                            Seated_Customer.Served()
                        ELSE
                            Player.Meals.REMOVE(Player.Meals[0])
                            Seated_Customer.ServedWrong()
                        END IF
                    END IF
                    index <- index + 1
                END FOR
            END IF
        END FOR
    END METHOD
END CLASS
