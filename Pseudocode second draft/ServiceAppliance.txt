DECLARE CLASS ServiceAppliance EXTENDS MonoBehaviour
    DECLARE PUBLIC GAMEOBJECT ApplianceTexture

    DECLARE PRIVATE BOOLEAN Loaded

    DECLARE PUBLIC VOID Load() DO
        IF Loaded THEN
            RETURN
        END IF

        INSTANTIATE(ApplianceTexture, GameConstants.SERVICE_APPLIANCE_POSITION, CURRENT_ROTATION)
        Loaded <- TRUE
    END METHOD

    DECLARE PUBLIC VOID GarbageCollect() DO
        IF Loaded IS FALSE THEN
            RETURN
        END IF

        DESTROY(ApplianceTexture)
    END METHOD

    DECLARE PRIVATE VOID OnStart() DO
        Loaded <- FALSE
    END METHOD

    DECLARE PRIVATE VOID Update() DO
    END METHOD
END CLASS
