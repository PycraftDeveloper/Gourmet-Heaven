// This object has tag "CurrentLevel"

ECLARE CLASS LevelOne EXTENDS MonoBehaviour
    DECLARE PUBLIC GAMEOBJECT Kitchen // only unload resources between scenes
    DECLARE PUBLIC GAMEOBJECT Restaurant // only unload resources between scenes
    DECLARE PUBLIC GAMEOBJECT Player
    DECLARE PUBLIC GAMEOBJECT Customer
    DECLARE PUBLIC GAMEOBJECT LevelOver

    DECLARE PUBLIC LIST OF GAMEOBJECTs Customers

    DECLARE PRIVATE FLOAT Customer_Spawn_Timer
    DECLARE PUBLIC FLOAT Level_Run_Time
    DECLARE PUBLIC FLOAT Level_Score

    DECLARE PRIVATE INT Customer_Next_Spawn_Time
    DECLARE PUBLIC INT Customers_In_Scene
    DECLARE PUBLIC INT Level
    DECLARE PUBLIC INT Sub_Level

    DECLARE PRIVATE VOID OnLevelEnd() DO
        DESTROY(Kitchen)
        DESTROY(Restaurant)
        DESTROY(Player)
        DESTROY(Customers)
        DESTROY(THIS GAMEOBJECT)
    END METHOD

    DECLARE PRIVATE VOID OnStart() DO
        INSTANTIATE(Kitchen, CURRENT_POSITION, CURRENT_ROTATION)
        INSTANTIATE(Restaurant, CURRENT_POSITION, CURRENT_ROTATION)
        INSTANTIATE(Player, NEW VECTOR3(0, 0, 0), CURRENT_ROTATION) // centred
        Kitchen.Load()
        Location <- GameConstants.KITCHEN
        Customer_Spawn_Timer <- 0
        Customer_Spawn_Time <- RANDOM_RANGE(GameConstants.CUSTOMER_SPAWN_TIME_MINIMUM, GameConstants.CUSTOMER_SPAWN_TIME_MAXIMUM)
        Sub_Level <- GameConstants.PART_ONE
        Level <- GameConstants.LEVEL_ONE
        Level_Score <- 0
    END METHOD

    DECLARE PRIVATE VOID Update() DO
        Level_Run_Time <- Level_Run_Time + DELTATIME

        IF Level_Run_Time > GameConstants.LEVEL_DURATION / 2 THEN
            IF Level_Run_Time >= GameConstants.LEVEL_DURATION THEN
                NEW GAMEOBJECT LevelOverMenu <- INSTANTIATE(LevelOver, CURRENT_POSITION, CURRENT_ROTATION)
                LevelOverMenu.Score <- Level_Score
                OnLevelEnd()
            ELSE
                Sub_Level <- GameConstants.PART_TWO
            END IF
        END IF

        IF Player IN COLLISION WITH RESTAURANT PORTAL THEN
            Kitchen.GarbageCollect()
            Restaurant.Load()
            MOVE PLAYER TO TOP OF FRAME

            FOR Existing_Customer IN Customers DO
                IF Existing_Customer.Location IS GameConstants.KITCHEN THEN
                    Existing_Customer.GarbageCollect() // unload customers resources when in different scene
                END IF
            END FOR
        ELSE IF Player IN COLLISION WITH KITCHEN PORTAL THEN
            Restaurant.GarbageCollect()
            Kitchen.Load()
            MOVE PLAYER TO BOTTOM OF FRAME

            FOR Existing_Customer IN Customers DO
                IF Existing_Customer.Meal == "" THEN
                    Level_Score <- Level_Score + Existing_Customer.Customer_Satisfaction
                    DESTROY(Existing_Customer) // I promise this isn't as bad as it sounds...
                    Customers.REMOVE(Existing_Customer)
                    Customers_In_Scene <- Customers_In_Scene - 1
                ELSE
                    IF Existing_Customer.Location IS RESTAURANT THEN
                        Existing_Customer.GarbageCollect() // unload customers resources when in different scene
                    END IF
                END IF
            END FOR
        END IF

        IF Customer_Spawn_Timer > Customer_Next_Spawn_Time THEN
            Customer_Next_Spawn_Timer <- 0
            Customer_Spawn_Time <- RANDOM_RANGE(GameConstants.CUSTOMER_SPAWN_TIME_MINIMUM, GameConstants.CUSTOMER_SPAWN_TIME_MAXIMUM)
            IF Customers_In_Scene < GameConstants.MAX_CUSTOMERS THEN
                INSTANTIATE(Customer, GameConstants.CUSTOMER_INITIAL_POSITION, CURRENT_ROTATION)
            END IF
        END IF
    END METHOD
END CLASS
