// This object has tag "Player"

DECLARE CLASS Player EXTENDS MonoBehaviour
    DECLARE PUBLIC GAMEOBJECT PlayerTexture
    DECLARE PUBLIC GAMEOBJECT UserJoystickInput

    DECLARE PUBLIC LIST OF STRINGs Meals

    DECLARE PRIVATE VOID OnStart() DO
        INSTANTIATE(PlayerTexture, CURRENT_POSITION, CURRENT_ROTATION)
        INSTANTIATE(UserJoystickInput, JOYSTICK_POSITION, CURRENT_ROTATION)
    END METHOD

    DECLARE PRIVATE VOID Update() DO
        NEW VECTOR3 DistanceMoved <- UserJoystickInput.NORMALISED * PLAYER_MOVEMENT_SPEED * DELTATIME
        CURRENT_POSITION <- CURRENT_POSITION + DistanceMoved

        PlayerTexture.POSITION <- CURRENT_POSITION
    END METHOD
END CLASS
