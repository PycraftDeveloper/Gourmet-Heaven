DECLARE CLASS LevelOverMenu EXTENDS MonoBehaviour
    DECLARE PUBLIC GAMEOBJECT BackgroundTexture
    DECLARE PUBLIC GAMEOBJECT GameLogo

    DECLARE PUBLIC GAMEOBJECT MainMenuButton
    DECLARE PUBLIC GAMEOBJECT QuitButton

    DECLARE PUBLIC GAMEOBJECT MainMenu

    DECLARE PUBLIC FLOAT Score

    DECLARE PRIVATE VOID OnMenuChange() DO
        DESTROY(BackgroundTexture)
        DESTROY(GameLogo)
        DESTROY(MainMenuButton)
        DESTROY(QuitButton)
        DESTROY(THIS GAMEOBJECT)
    END METHOD

    DECLARE PRIVATE VOID OnStart() DO
        INSTANTIATE(BackgroundTexture, CURRENT_POSITION, CURRENT_ROTATION)
        INSTANTIATE(GameLogo, LOGO_POSITION, CURRENT_ROTATION)
        INSTANTIATE(MainMenuButton, MAIN_MENU_BUTTON_POSITION, CURRENT_ROTATION)
        INSTANTIATE(QuitButton, QUIT_BUTTON_POSITION< CURRENT_ROTATION)
    END METHOD

    DECLARE PRIVATE VOID Update() DO
        DISPLAY SCORE

        IF LevelOneButton PRESSED THEN
            INSTANTIATE(MainMenu, CURRENT_POSITION, CURRENT_ROTATION)
            OnMenuChange()
        ELSE IF LevelTwoButton PRESSED THEN
            QUIT APPLICATION
        END IF
    END METHOD
END CLASS
